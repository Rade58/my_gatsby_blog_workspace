  allFile(filter: {internal: {mediaType: {eq: "image/svg+xml"}}, relativePath: {eq: $relPath}}) {
  	
  	nodes {
      absolutePath
      relativePath
      
      # OVO SAM SAMO OSTAVIO DA TI KAZEM DA JE MOGUCE QUERY-EOVATI
